8.4 Use an external vector

Rename with an external vector of names

# Uses a named vector...
lookup <- c(`Left-right scale` = "leftrigh", 
            `Libertarian-authoritarian scale` = "libauth", 
            `Welfarism scale` = "welfare2", 
            dole = "Dole"
            )

rename(df, all_of(lookup)) %>% names
