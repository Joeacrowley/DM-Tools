3.2 Mean of multiple variables at once (each mean is separate)

3.2.1 A purrr-style formula

df <- df %>% mutate(

3.3 Calculate multiple statistics per variable

3.3.1 Requires a named list of functions

Adds a column with the mean of welfare2 and the standard deviation.

df %>% select(welfare2) %>% 
  mutate(
    across(
      all_of("welfare2"), 
      # First part is name for stat column, second is the statistic function
      list(var_mean = mean, sd = sd), 
      .names = "{.col}_{.fn}")
    )

3.5.2.1 Another example

set.seed(123)  # for reproducibility

num_df <- data.frame(
  var1 = sample(1:100, 50, replace = T),
  var2 = sample(1:45, 50, replace = T),
  var3 = sample(1:20, 50, replace = T),
  var4 = sample(1:75, 50, replace = T),
  var5 = sample(1:50, 50, replace = T)
)

num_df <- num_df %>% mutate(
  
  # All columns
  row_total = rowSums(across(everything()), na.rm = T), 
  
  # Identify column numbers
  rows_2_4 = rowSums(across(c(2,4)), na.rm = T), 
  
  # Identfy variable with ends_with
  ends_with_3_5 = rowSums(across(ends_with(c("3","5")))),
  
  # With character vector
  `1_and_2` = rowSums(across(all_of(c("var1", "var2"))))
  
)
