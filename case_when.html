# convenience function to run count() for all of a vector of variables
count_df <- function(df, vars) {
  map(vars, 
      ~df %>%
        mutate_at(.x, ~as_factor(.x, levels = "both")) %>%
        count(across(all_of(.x))) %>% 
        mutate(var = .x) %>% 
        rename(cat = .x) %>% 
        relocate(var)
      ) %>%
    bind_rows
}

check_recode <- function(df, vars, suffix) {
  for (var in vars) {
    new_var <- paste0(var, suffix)
    cat("\n===", var, "===========================\n")
    result <- count(df, .data[[var]], .data[[new_var]])
    print(result, n = nrow(result))
    cat("\n")
  }
}
